# GTO Wizard Auto-Login Credentials Example

# Method 1: Using Environment Variables (Recommended)
export GTO_EMAIL="your_apple_id@privaterelay.appleid.com"
export GTO_PASSWORD="your_password"

# Then run:
python3 token_grabber_selenium.py --update-scraper

# Method 2: Using Command Line Arguments
python3 token_grabber_selenium.py \
  --email "your_email@example.com" \
  --password "your_password" \
  --update-scraper

# Method 3: Create a .env file (add to .gitignore!)
# Create a file named .env with:
GTO_EMAIL=your_apple_id@privaterelay.appleid.com
GTO_PASSWORD=your_password

# SECURITY NOTE:
# - NEVER commit credentials to git
# - Add .env to .gitignore
# - Use environment variables for CI/CD
# - Consider using a password manager
